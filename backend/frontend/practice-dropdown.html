// Single select dropdown
WebElement toolDropdown = driver.findElement(By.id("singleSelectDropdown"));
Select toolSelect = new Select(toolDropdown);
toolSelect.selectByVisibleText("Selenium WebDriver");

// Multi select dropdown
WebElement skillsDropdown = driver.findElement(By.id("multiSelectDropdown"));
Select skillsSelect = new Select(skillsDropdown);
skillsSelect.selectByVisibleText("Java");
skillsSelect.selectByVisibleText("Selenium");

// Read selected options
List<WebElement> selectedOptions = skillsSelect.getAllSelectedOptions();
for (WebElement option : selectedOptions) {
    System.out.println(option.getText());
}
