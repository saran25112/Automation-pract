// Import required Selenium classes
import org.openqa.selenium.*;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.*;
import java.time.Duration;
import java.util.List;

public class ChromeLaunch {

    // WebDriver instance
    private static WebDriver driver;

    // JavaScript Executor for advanced actions
    private static JavascriptExecutor js;

    // Explicit wait
    private static WebDriverWait wait;

    public static void main(String[] args) {

        // Launch Chrome browser
        driver = new ChromeDriver();
        driver.manage().window().maximize();

        // Open the calendar page
        driver.get("practice-calendar");

        js = (JavascriptExecutor) driver;
        wait = new WebDriverWait(driver, Duration.ofSeconds(10));

        // Click Date of Birth calendar button
        WebElement dobCalander = driver.findElement(By.id("dobDateTrigger"));
        clickElement(dobCalander);

        // Select Year from dropdown
        WebElement years = driver.findElement(By.id("compactCalendarYearSelect"));
        selectByText(years, "2002");

        // Select Month (index based)
        WebElement months = driver.findElement(By.id("compactCalendarMonthSelect"));
        selectByIndex(months, 11);

        // Select Date from calendar
        List<WebElement> days = driver.findElements(
            By.xpath("//button[contains(@class,'compact-calendar-day')]")
        );
        selectDate(days, "25");
    }

    // Click element safely (handles scroll + interception)
    public static void clickElement(WebElement element) {
        wait.until(ExpectedConditions.visibilityOf(element));
        wait.until(ExpectedConditions.elementToBeClickable(element));

        // Scroll element to center of screen
        js.executeScript("arguments[0].scrollIntoView({block:'center'});", element);

        try {
            element.click();
        } catch (Exception e) {
            // Fallback using JavaScript click
            js.executeScript("arguments[0].click();", element);
        }
    }

    // Select dropdown using visible text
    public static void selectByText(WebElement element, String text) {
        Select select = new Select(element);
        select.selectByVisibleText(text);
    }

    // Select dropdown using index
    public static void selectByIndex(WebElement element, int index) {
        Select select = new Select(element);
        select.selectByIndex(index);
    }

    // Select a specific date from calendar
    public static void selectDate(List<WebElement> days, String date) {

        for (WebElement day : days) {
            if (day.getText().equals(date)) {

                // Click using JS to avoid interception issues
                js.executeScript("arguments[0].click();", day);
                break;
            }
        }
    }
}
⚠️ Important
using JavaScript click for calendar button , click may not work
explicit wait and scroll element to view or optional.
