Java Selenium 
// Get all rows from the table body
List<WebElement> rows = driver.findElements(
 By.xpath("//table[@id='employeeTable']/tbody/tr")
);

int rowCount = rows.size();
int colCount = rows.get(0).findElements(By.tagName("td")).size();

// Create a 2D array based on row count and 4 columns
String[][] tableData = new String[rowCount][colCount];

// Use nested loops to read every cell
for (int i = 0; i < rows.size(); i++) {
 List<WebElement> cols = rows.get(i).findElements(By.tagName("td"));

 for (int j = 0; j < cols.size(); j++) {
 tableData[i][j] = cols.get(j).getText();
 }
}

// Print the full table data
for (int i = 0; i < tableData.length; i++) {
 for (int j = 0; j < tableData[i].length; j++) {
 System.out.print(tableData[i][j] + " | ");
 }
 System.out.println();
}
