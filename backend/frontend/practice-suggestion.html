// Type partial text in the suggestion box
driver.findElement(By.id("suggestionInput")).sendKeys("sel");

// Wait for suggestion buttons to appear
WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(5));
List<WebElement> suggestions = wait.until(
    ExpectedConditions.visibilityOfAllElementsLocatedBy(
        By.cssSelector("#suggestionResults button")
    )
);

// Click the matching suggestion
for (WebElement item : suggestions) {
    if (item.getText().equals("Selenium WebDriver")) {
        item.click();
        break;
    }
}
