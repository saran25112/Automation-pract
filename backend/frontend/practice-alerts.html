Java Selenium 
// Trigger and handle Simple Alert
driver.findElement(By.id("simpleAlertBtn")).click();
Alert alert = driver.switchTo().alert();
String alertText = alert.getText();
alert.accept();

// Handle Confirm Dialog
driver.findElement(By.id("confirmAlertBtn")).click();
Alert confirm = driver.switchTo().alert();
confirm.accept(); // Click OK
// confirm.dismiss(); // Click Cancel

// Handle Prompt Dialog
driver.findElement(By.id("promptAlertBtn")).click();
Alert prompt = driver.switchTo().alert();
prompt.sendKeys("My Name");
prompt.accept();
