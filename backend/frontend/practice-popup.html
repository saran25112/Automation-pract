Java Selenium 
// Old flow: open popup directly
driver.findElement(By.id("openModalBtn")).click();

// Get popup text
String popupText = driver.findElement(By.id("modalMessage")).getText();
System.out.println(popupText);

// Close popup
driver.findElement(By.id("modalCloseBtn")).click();


// New flow: trigger the random popup flow
driver.findElement(By.id("openRandomPopupBtn")).click();

// Wait until either popup text or fallback status appears
WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(5));
wait.until(webDriver -> {
 boolean popupVisible = driver.findElement(By.id("practiceModal")).getAttribute("class").contains("show");
 String status = driver.findElement(By.id("popupStatus")).getText();
 return popupVisible || status.contains("did not appear");
});

// If the popup appeared, get text and close it
if (driver.findElement(By.id("practiceModal")).getAttribute("class").contains("show")) {
 String randomPopupText = driver.findElement(By.id("modalMessage")).getText();
 System.out.println(randomPopupText);
 driver.findElement(By.id("modalCloseBtn")).click();
}
