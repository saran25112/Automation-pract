Java Selenium 
// Get visible text
String text = driver.findElement(By.id("visibleTextContent")).getText();

// Get input value
String value = driver.findElement(By.id("attributeInput")).getAttribute("value");

// Get custom data attribute
String data = driver.findElement(By.id("attributeInput")).getAttribute("data-custom");

// Reveal hidden text and read it
driver.findElement(By.id("toggleHiddenBtn")).click();
WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(5));
WebElement hidden = wait.until(
 ExpectedConditions.visibilityOfElementLocated(By.id("hiddenText"))
);
System.out.println(hidden.getText());

// Get paragraph text
String para = driver.findElement(By.id("paragraphText")).getText();
